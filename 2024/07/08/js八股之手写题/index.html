function throttle(f, time) {
  let startTime = 0;
  return function() {
    const endTime = new Date();
    let args = argument;
    if(endTime - startTime > time) {
      f.apply(this, args);
      startTime = endTime;
    }
  }
}

function clone(target, map = new WeakMap()) {
  if(typeof target !== 'null' && (typeof target === 'object' || typeof target === 'function')) {
    const cloneTarget = isArray(target) ? [] : {};

    if(map.get(target)) return map.get(target);//解决循环引用
    map.set(target, cloneTarget);

    for(const key in target) {
      if(target.hasOwnProperty(key))
      cloneTarget(key) = clone(target(key), map)
    }
    return cloneTarget
  }
  else return target
}

function _instanceof(left, right) {
  let rightProto = right.prototype;
  let leftProto = left.__proto__;
  while(1) {
    if(leftProto === null) return false;
    if(leftProto = rightProto) return true;
    leftProto = leftProto.__proto__;
  }
}

function flat(arr) {
  const res = [];
  const stack = [...arr];
  while(stack.length) {
    const item = stack.pop();
    if(Array.isArray(item)) stack.push(...item);
    else res.push(item);
  }
  return res.reverse();
}

call apply bind

call apply
相当于在目标对象上创建一个临时方法
（边界判断，使用symbol， 绑定this，用后即删）

Function.prototype.myCall(target, ...args) {//apply args
  target = target || window;
  const symbolKey = Symbol();
  target[symbolKey] = this;
  const res = target[symbolKey](...args);
  delete target[symbolKey];
  return res;
}

bind
返回一个函数

Function.prototype.myBind(target, ...outerArgs) {
  target = target || {};
  const symbolKey = Symbol();
  target[symbolKey] = this;
  return function(...innerArgs) {
    const res = target[symbolKey](...outerArgs, ...innerArgs);
    return res;
  }
}

函数柯里化

把一个函数分解成多个子函数
(bind运用了柯里化思想)

  • 参数定长 （如 fn = (a,b,c) => a+b+c）
function curry(fn, ...presetArgs) {
  const argLen = fn.length //求函数参数长度
  return function(...restArgs) {
    const allArgs = [...presetArgs, ...restArgs];
    if(allArgs.length >= argLen) {
      return fn.call(this, ...allArgs);
    }
    else {
      return curry.call(null, fn, ...allArgs);
    }
  } 
}
  • 参数不定长
    因为没法判断边界条件，所以改写toString方法
function curry(fn, ...presetArgs) {
  function curried(...restArgs) {
    const allArgs = [...presetArgs, ...restArgs];
    return curry.call(null, fn, ...allArgs);
  }
  curried.toString = () => {
    return fn.call(this, ...presetArgs);
  }

  return curried;
}
