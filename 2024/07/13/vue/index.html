响应式原理

Vue2 中使用Object.defineProperty() 实现

function reactive(obj, key, value) {
  Object.defineProperty(data, key, {
    get() {
      console.log(`访问了${key}属性`)
      return value;
    },
    set(val) {
      console.log(`将${key}由->${value}->设置成->${val}`)
      if (value !== val) {
        value = val;
      }
    }
  })
}

缺点：只能侦听对象的已有属性，对新增的属性无效

Vue3改为Proxy + Reflect

function reactive(target) {
    const handler = {
        get(target, key, receiver) {
            track(receiver, key)
            return Reflect.get(target, key, receiver)
        },
        set(target, key, value, receiver) {
            Reflect.set(target, key, value, receiver)
            trigger(receiver, key)
        }
    }

    return new Proxy(target, handler)
}

对传入reactive的target，将其包装成proxy，对get和set操作进行拦截
（Reflect.get相当于receiver[key] .set相当于receiver[key] = value）
使用reflect的优点：和Proxy搭配提高语义化，且应该把操作放在receiver（代理化的对象）上

track:依赖收集

const targetMap = new WeakMap()//存储所有对象
let activeEffect = null

function track(target, key) {
    if (!activeEffect) return//避免如console操作
    let depsMap = targetMap.get(target)//存储单个对象的所有属性
    if (!depsMap) {
        targetMap.set(target, depsMap = new Map())
    }

    let dep = depsMap.get(key)//存储属性的所有依赖
    if (!dep) {
        depsMap.set(key, dep = new Set())
    }
    dep.add(activeEffect) 
}//weakMap -> Map -> Set

function effect(fn) {//设置activeEffect为当前effect，使用完清空
    activeEffect = fn
    activeEffect()
    activeEffect = null
}

trigger: 发布更新

function trigger(target, key) {
    let depsMap = targetMap.get(target)
    if (depsMap) {
        const dep = depsMap.get(key)
        if (dep) {
            dep.forEach(effect => effect())
        }
    }
}

之后可以基于此实现ref和computed:

function ref(initValue) {//即只有value属性的reactive
    return reactive({
        value: initValue
    })
}
function computed(fn) {//内部是一个effect
    const result = ref()
    effect(() => result.value = fn())
    return result
}
